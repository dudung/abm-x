   \setbox0=\hbox{$\displaystyle{#1}$}%
   \dimen0=\ht0 \advance\dimen0 by\dp0
   \setbox1=\hbox{$\vcenter{\copy0}$}%

      $\textfont3=\tmp \displaystyle\int_{\fxintL}^{\fxintU}$}

   \endgroup
